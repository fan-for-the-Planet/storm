async def process_media(event, media_type: str, suffix: str, mime_type: str, prompt: str):  
    """–û–±—Ä–∞–±–æ—Ç–∫–∞ –º–µ–¥–∏–∞-–∫–æ–Ω—Ç–µ–Ω—Ç–∞ (–≤–∏–¥–µ–æ/—Ñ–æ—Ç–æ)"""

        # –ü—Ä–æ–≥—Ä–µ—Å—Å —Å–∫–∞—á–∏–≤–∞–Ω–∏—è  
        last_progress = time.time()  
  
        def progress_callback(current, total):  
            nonlocal last_progress  
            now = time.time()  
            if now - last_progress > 10:  
                logger.info(f"–°–∫–∞—á–∏–≤–∞–Ω–∏–µ {media_type}: {current}/{total} –±–∞–π—Ç ({current / total:.1%})")

        await event.message.download_media(  
            file=media_path,  
            progress_callback=progress_callback  
        )  
        logger.info(f"{media_type.capitalize()} —Å–∫–∞—á–∞–Ω: {media_path} ({os.path.getsize(media_path)} –±–∞–π—Ç)")

    finally:  
        # –£–¥–∞–ª—è–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π —Ñ–∞–π–ª
